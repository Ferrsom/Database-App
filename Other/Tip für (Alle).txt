public List<Futterzu> getFutterZuordnung(int tiernr)
{

    MySqlCommand comm = CreateCommand(conn);


comm.CommandText = "SELECT * FROM FutterTier " 
    if(tiernr != -1 )
        comm.commandText += "WHERE tierid = " + tiernr.toString();      
    comm.commandtext += " ORDER BY Tier;";
}
